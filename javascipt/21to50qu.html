Q21: What are the looping structures in JavaScript? Any one Example?

In JavaScript, there are several looping structures that you can use to iterate over code. 1. for Loop: Repeats a block of code a specified number of times. 2. while Loop: Repeats a block of code as long as a specified condition is true. 3. do...while Loop: Similar to the while loop, but guarantees that the code block. executed at least once before the condition is tested. 5. for...in Loop: Iterates over the enumerable properties of an object. 6. for...of Loop: Iterates over the values of iterable objects (like arrays or strings). const numbers = [1, 2, 3, 4, 5]; for (let i = 0; i < numbers.length; i++) { console.log(numbers[i]); }

Q33: Write a JavaScript program to compare two objects?

const obj1 = { name: "Alice", age: 25, address: { city: "Wonderland", zip: "12345" } }; const obj2 = { name: "Alice", age: 25, address: { city: "Wonderland", zip: "12345" } }; const obj3 = { name: "Bob", age: 30, address: { city: "Builderland", zip: "67890" } }; console.log(deepEqual(obj1, obj2)); // Output: true console.log(deepEqual(obj1, obj3)); // Output: false function deepEqual(obj1, obj2) { if (obj1 === obj2) return true; if (typeof obj1 !== 'object' || obj1 === null || typeof obj2 !== 'object' || obj2 === null) { return false; } const keys1 = Object.keys(obj1); const keys2 = Object.keys(obj2); if (keys1.length !== keys2.length) return false; for (let key of keys1) { if (!keys2.includes(key) || !deepEqual(obj1[key], obj2[key])) { return false; } } return true; }

Q34: Write a JavaScript program to convert an array of objects into CSV string?

const data = [ { name: "Alice", age: 25, city: "Wonderland" }, { name: "Bob", age: 30, city: "Builderland" }, { name: "Charlie", age: 35, city: "Chocoland" } ]; const csvString = arrayToCSV(data); console.log(csvString); function arrayToCSV(array) { if (array.length === 0) return ''; const headers = Object.keys(array[0]); const csvRows = [headers.join(',')]; for (const obj of array) { const values = headers.map(header => { const value = obj[header] !== undefined ? obj[header] : ''; return "${String(value).replace(/"/g, '""')}"; }); csvRows.push(values.join(',')); } return csvRows.join('\n'); }
